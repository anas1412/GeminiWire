Getting Started

Installation


# Clone the repository
git clone https://github.com/anas1412/GeminiWire.git
cd GeminiWire

# Install dependencies
pip install -r requirements.txt

# Configure API Key
cp .env.example .env
# Add your Gemini API key in .env

# Start the service
uvicorn server:app --reload

    Generate a brief anime character description

SQL Query Generation 


def fetch_users_with_skill(inputs: dict):
    """
    Generate SQL query to fetch users with a specific skill
    """
    skill = inputs.get('skill')
    return f"SELECT * FROM users WHERE skills LIKE '%{skill}%'"

# Example Usage
sql_query = execute_function("fetch_users_with_skill", {
    "skill": "Coding"
})

Complex Workflow Example 


# Chaining multiple AI-powered functions
character1 = execute_function("generate_character", {
    "name": "Alice", "gender": "female", "age": 22
}).data

character2 = execute_function("generate_character", {
    "name": "Anas", "gender": "male", "age": 27
}).data

plot = "Coding an AI framework that integrates dynamic function execution"

world = execute_function("generate_world", {
    "plot": plot, 
    "characters": [character1, character2]
}).data
